    print " Loto Çekiliş Programına Hoşgeldiniz!!!! "

secenek2=" (2) Süper Loto "
secenek3=" (3) On numara "

    print secenek1
print secenek2

print
print "Çıkış için 'q' yazınız."

sayisal_loto=random.sample(range(1,49), 6)
sayisal_loto.sort()
print

super_loto.sort()

    if secim=="3":
on_numara=random.sample(range(1,80), 22)
on_numara.sort()

    if secim=="4":

if secim=="q":
